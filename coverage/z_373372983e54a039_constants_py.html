1""" 

2Physical constants and conversion factors used in the package. 

3 

4The values given here have been taken from the

12m_n = 1.674_927_498_04e-27 # kg neutron mass

14# some conversion constants used 

15kilo = 1e3

17fm2angstrom = 1e-5 

18eV2J = h_Js / h_eVs 

19 

20r_e = 2.817_940_3262 # fm - classical electron radius 

21r_e_angstrom = fm2angstrom * r_e # Å - classical electron radius 

22sigma_to_b_1A = 5.0e-4 # fm/barn absorption at 1.0Å (sigma=4*pi*b/k) 

23wavelength_for_b = ( 

24 h_Js / m_n / 2200.0 * m2angstrom 

25) # 1.798 Å (wavelength at 2200 m/s, the default for reported b-values) 

26sigma_to_b = sigma_to_b_1A / wavelength_for_b # fm/barn absorption at 1.798Å (sigma=4*pi*b/k)
